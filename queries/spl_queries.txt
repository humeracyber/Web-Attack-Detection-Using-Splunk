# 1. Total HTTP Events
index=main sourcetype="http_custom_logs"
| stats count

---

# 2. Brute Force Login Attempts
index=main sourcetype="http_custom_logs" uri="/login.php" status=401
| stats count BY src
| sort -count

---

# 3. Successful Logins
index=main sourcetype="http_custom_logs" uri="/login.php" status=200
| table _time src method uri
| sort _time

---

# 4. SQL Injection Attempts
index=main sourcetype="http_custom_logs"
| regex uri="(\bOR\b|\bUNION\b|\bSELECT\b|;|')"
| table _time src uri status
| sort _time

---

# 5. Sensitive File Access (.zip, .db)
index=main sourcetype="http_custom_logs"
| where like(uri, "%.zip") OR like(uri, "%.db")
| table _time src uri status

---

# 6. Suspicious File Uploads
index=main sourcetype="http_custom_logs"
| where like(uri, "%upload.php")
| table _time src method uri file status
| sort _time

---

# 7. Web Shell / Command Execution
index=main sourcetype="http_custom_logs"
| where like(uri, "%cmd=%")
| table _time src uri status
| sort _time

---

# 8. Suspicious Activity Timeline
index=main sourcetype="http_custom_logs"
| eval suspicious=if(
    status=401 
    OR like(uri,"%cmd=%") 
    OR like(uri,"%.zip") 
    OR like(uri,"%.db") 
    OR match(uri,"(OR|UNION|SELECT|;)"),
    "yes","no"
)
| where suspicious="yes"
| timechart count BY src
